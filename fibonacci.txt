#include <iostream>
using namespace std;

int fibonacci(int n) {
    if (n <= 1)
        return n;
    
    return fibonacci(n - 1) + fibonacci(n - 2);
}

int main() {
    int num;
    cout << "Ingrese un nÃºmero para calcular su serie de Fibonacci: ";
    cin >> num;

    cout << "Serie de Fibonacci hasta " << num << ":" << endl;
    for (int i = 0; i <= num; i++) {
        cout << fibonacci(i) << " ";
    }
    
    return 0;
}
